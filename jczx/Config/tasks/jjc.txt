/ 竞技场日常 - 模拟军演
/ 依赖: MainMenu 中的 goto-home, auto-fight, click-get-item, click-center, wait-*, in_location 等公共实体

[jjc-simulate]
type: task
view: on
name: 竞技场日常
action: goto-jjc,condition-simulate-need-fight
settings: settings-jjc-simulate
log: +28挑战%{times_28}次，+26挑战%{times_26}次，挑战失败%{times_loss}次

[settings-jjc-simulate]
type: settings
fields: setting-power-threshold-jjc

[setting-power-threshold-jjc]
type: setting
setting_type: integer
label: 战力阈值
default: 50000
min: 0

[jjc-simulate-values]
setting-power-threshold-jjc: 25000

[goto-jjc]
type: task
name: 前往竞技场
action: goto-home,click-fight,click-simulate,click-jjc

[click-simulate]
type: click
name: 模拟军演
target: buttons\simulatedMilitaryExercises.png
sleep: 2

[click-jjc]
type: click
name: 竞技场
target: buttons\competition.png
sleep: 3

[condition-simulate-need-fight]
type: condition
name: 条件-是否挑战
desc: 条件挑战次数 大于0,
condition: &{get-simulate-times > 0 & get-simulate-refresh-times > 0}
condition_then: get-average-refresh-times,condition-to-fight-jjc
condition_else: goto-home

[get-average-refresh-times]
type: task
name: 计算可以平均使用刷新的次数 -> average_times
action: get-simulate-times,get-simulate-refresh-times,get-combat-power,context-simulate-fight-refresh-times,context-simulate-continue-times,context-simulate-continue-times-max-3

/ 竞技场流程:
/   | 获取(挑战次数,获取第一位的战力值,获取剩余刷新次数,获取是+28/+26)
/ 计算平均使用刷新次数 - times
/   | 
/ 判断是否小于战力阈值 & (times是否<=0 | 是+28挑战)  <----------------\
/   | Y             |                                              |
/ 发起挑战           \---------------N-----------> 点击刷新，times-1/
/
[condition-to-fight-jjc]
type: condition
name: 条件判断是否挑战
desc: 战力阈值 & （平均刷新次数期望 <= 0 | 是+28）,不满足条件继续刷新
condition: &{%{combat_power} <= ${jjc-simulate-values:setting-power-threshold-jjc} & (%{average_times} <= 0 | @{condition-jjc-in-28})}
condition_then: click-jjc-fight,click-jjc-fight-sure,auto-fight,click-get-item,condition-compare-jjc-fight-times,context-check,wait-2,condition-simulate-need-fight
condition_else: context-average-times-1,click-jjc-fight-refresh,get-combat-power,get-simulate-refresh-times,get-simulate-times,condition-simulate-need-fight-1

[condition-simulate-need-fight-1]
type: condition
name: 条件-是否挑战
desc: 条件挑战次数 大于0,
condition: &{get-simulate-times > 0 & get-simulate-refresh-times > 0}
condition_then: condition-to-fight-jjc
condition_else: goto-home

[click-jjc-fight-refresh]
type: click
name: 刷新
target: buttons\competitionCancel.png
sleep: 1

[click-jjc-fight]
type: click
name: 挑战
target: buttons\jjcScore.png

[condition-compare-jjc-fight-times]
type: condition
name: 计算挑战状态
condition: &{%{win_flag}}
condition_then: condition-compare-jjc-fight-times-1
condition_else: context-times_loss-add-1

[condition-compare-jjc-fight-times-1]
type: condition
name: 计算挑战状态, 获胜状态下区分 28/26
condition: &{%{is_28}}
condition_then: context-times_28-add-1
condition_else: context-times_26-add-1

[context-times_26-add-1]

context_key: times_26

[context-times_loss-add-1]
